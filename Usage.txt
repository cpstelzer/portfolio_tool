# Install dependencies
conda env create -f environment.yml

# Basic run (buy-and-hold benchmark, default VWCE.DE)
python portfolio_model.py transaction.csv isin_to_yahoo.csv

# Mirror-flows mode with custom benchmark
python portfolio_model.py transaction.csv isin_to_yahoo.csv --benchmark IWDA.AS --buy_and_hold false

# Restrict date range and export results
python portfolio_model.py transaction.csv isin_to_yahoo.csv \
    --benchmark VWCE.DE \
    --start_date 2022-01-01 \
    --end_date 2024-12-31 \
    --export_csv results.csv
